# SME Pipeline Methods Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add setter generics and `SpaceMarkersExperiment`-dispatched methods for every per-step pipeline function, so the step-by-step vignette can chain SME-in/SME-out calls instead of poking slots directly.

**Architecture:** Each pipeline function (e.g., `find_all_hotspots`, `get_pairwise_interacting_genes`, `calculate_overlap_undirected`, etc.) becomes an S4 generic with two methods: a default method containing the existing function body (signature unchanged — no back-compat break), and a method dispatching on `"SpaceMarkersExperiment"` that reads inputs via existing accessors, delegates to the default method, and writes the output via a matching new setter generic.

**Tech Stack:** R / Bioconductor, S4, `SpatialExperiment` superclass, `testthat`, `devtools::document()`/`devtools::test()`, `rmarkdown::render`.

**Spec:** [`docs/superpowers/specs/2026-04-20-sme-pipeline-methods-design.md`](../specs/2026-04-20-sme-pipeline-methods-design.md)

## Shared patterns

**Generic signature convention.** Each new generic keeps the exact first-arg name of the current function. Dispatch happens on that first arg. All pre-refactor args remain in the generic's signature (preserving back-compat for positional callers).

**Default method pattern.** Convert:
```r
find_all_hotspots <- function(spPatterns, params = NULL, ...) {
    # body
}
```
to:
```r
setMethod("find_all_hotspots", "data.frame",
    function(spPatterns, params = NULL, ...) {
        # body (unchanged)
    }
)
```
The roxygen `@export` on the original function is moved onto the `setGeneric` in `AllGenerics.R`. Roxygen comment blocks for the original function stay where they are (attached to the method) so documentation content is preserved.

**SME method pattern.** All SME methods follow extract → call default → write:
```r
setMethod("NAME", "SpaceMarkersExperiment", function(firstArg, ...) {
    sme <- firstArg
    # 1. validate prerequisite slots
    # 2. extract inputs via accessors / helpers
    # 3. call the default method with those inputs
    # 4. write result via matching setter
    sme
})
```

**Test file.** All new test blocks go in `tests/testthat/test-SpaceMarkersExperiment-pipeline.R`, appended to the end of the file. Each test block uses `test_that("...", { ... })`.

**Test fixture.** All tests build the SME the same way:

# NULL-coalescing helper (used by SME pipeline methods for optional-arg fallbacks)
`%||%` <- function(a, b) if (!is.null(a)) a else b

# Unexported helpers used by SME pipeline methods
.sme_expr <- function(sme) {
    if ("logcounts" %in% SummarizedExperiment::assayNames(sme))
        SummarizedExperiment::assay(sme, "logcounts")
    else
        SummarizedExperiment::assay(sme, 1L)
}

.sme_spPatterns <- function(sme) {
    coords <- as.data.frame(SpatialExperiment::spatialCoords(sme))
    pats <- as.data.frame(spatial_patterns(sme))
    df <- data.frame(barcode = colnames(sme),
                     coords, pats,
                     check.names = FALSE,
                     stringsAsFactors = FALSE)
    rownames(df) <- colnames(sme)
    df
}

#' @rdname hotspots
#' @aliases hotspots<-,SpaceMarkersExperiment-method
#' @param value A data.frame of hotspot assignments.
#' @export
setMethod("hotspots<-", "SpaceMarkersExperiment",
    function(x, type = c("undirected", "pattern", "influence"), value) {
        type <- match.arg(type)
        md <- S4Vectors::metadata(x)
        if (is.null(md$hotspots)) md$hotspots <- list()
        md$hotspots[[type]] <- value
        S4Vectors::metadata(x) <- md
        x
    }
)

#' @rdname interactions
#' @aliases interactions<-,SpaceMarkersExperiment-method
#' @param value A named list of per-pair interaction results.
#' @export
setMethod("interactions<-", "SpaceMarkersExperiment", function(x, value) {
    md <- S4Vectors::metadata(x)
    md$interactions <- value
    S4Vectors::metadata(x) <- md
    x
})

#' @rdname influence_map

#' @export
setMethod("influence_map<-", "SpaceMarkersExperiment", function(x, value) {

    md$influence <- value

})

#' @aliases find_all_hotspots,SpaceMarkersExperiment-method

setMethod("find_all_hotspots", "SpaceMarkersExperiment",
    function(spPatterns, params = NULL, outlier = "positive",
             nullSamples = 100, includeSelf = TRUE, ...) {
        sme <- spPatterns
        if (is.null(params)) params <- spatial_params(sme)
        hs <- find_all_hotspots(.sme_spPatterns(sme),
                                params = params, outlier = outlier,
                                nullSamples = nullSamples,

        hotspots(sme, type = "undirected") <- hs
        sme
    }
)

                minvals = minvals %||% default_min,
                maxvals = maxvals %||% default_max)

        hs <- find_hotspots_gmm(src, threshold = thr, ...)

    }

#' Filter an SME by gene expression / barcode match and (re)compute spatial params
#' @keywords internal
.apply_sme_filters <- function(sme, genes = NULL, min.gene.expr = 10,
                               spatialDir = "spatial",
                               pattern = "scalefactors_json.json",
                               sigma = NULL, threshold = 4,
                               resolution = "fullres") {
    message("Preparing data from SpaceMarkersExperiment...")
    # Barcode consistency
    expr <- .sme_expr(sme)
    pat_barcodes <- rownames(spatial_patterns(sme))
    if (!is.null(pat_barcodes)) {
        keep <- intersect(colnames(expr), pat_barcodes)
        if (length(keep) < ncol(sme)) {
            sme <- sme[, keep]
        }
    }
    # Gene filtering
    e <- .sme_expr(sme)
    keep_genes <- if (!is.null(genes))
        intersect(genes, rownames(e))
    else
        rownames(e)[which(apply(e, 1, sum) > min.gene.expr)]
    sme <- sme[keep_genes, ]
    # Spatial params — compute if missing
    if (is.null(spatial_params(sme))) {
        message("Computing optimal parameters...")
        stored_visiumDir <- sme@spacemarkers$params$visiumDir
        stored_res <- sme@spacemarkers$params$resolution %||% resolution
        op <- get_spatial_parameters(
            spatialPatterns = .sme_spPatterns(sme),
            visiumDir = stored_visiumDir %||% ".",
            spatialDir = spatialDir, pattern = pattern,
            sigma = sigma, threshold = threshold, resolution = stored_res)
        spatial_params(sme) <- op
    }
    sme
}
